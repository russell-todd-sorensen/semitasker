Implement Trie (Prefix Tree)

Implement a trie with insert, search, and 
startsWith methods.

Example:

Trie trie = new Trie();

trie.insert("apple");
trie.search("apple");   // returns true
trie.search("app");     // returns false
trie.startsWith("app"); // returns true
trie.insert("app");   
trie.search("app");     // returns true
Note:

You may assume that all inputs are consist of lowercase 
letters a-z. All inputs are guaranteed to be non-empty 
strings.


class Trie {
    root = {val:null};
    initialized = false;

    constructor () {
        this.initialized = true;
        return this;
    }

    insert(word) {

        let r = this.root,
            tmpRoot = r,
            len = word.length;

        for (let i=0,char;i<len;i++) {
            char = word[i];
            if (tmpRoot[char]) {
                tmpRoot = tmpRoot[char];
            }
            else {
                tmpRoot[char] = {val:null};
                tmpRoot = tmpRoot[char];
            }
        }

        if (tmpRoot["val"] == null) {
            tmpRoot["val"] = word;
        }
    };

    search (word) {
        let r = this.root,
            tmpRoot = r,
            len = word.length,
            found = false;

        for (let i=0,char;i<len;i++) {
            char = word[i];
            if (tmpRoot[char]) {
                tmpRoot = tmpRoot[char];
            } else {
                break;
            }
        }
        if (tmpRoot["val"] && tmpRoot["val"] == word) {
            found = true;
        }
        return found;
    };

    startsWith(prefix) {
        let r = this.root,
            tmpRoot = r,
            len = prefix.length,
            i   = 0,
            prefixFound = false;

        for (let char;i<len;i++) {
            char = prefix[i];
            if (tmpRoot[char]) {
                tmpRoot = tmpRoot[char];
            } else {
                break
            }
        }
        return (i==len?true:false);
    }

};
