'application'
 'name'Pg'
 'variable'
  'name'PgVersion'
  'default'7.3.4'
 ''
 'variable'
  'name'PgBaseUrl'
  'default'http://zmbh.com/app-install/postgresql/'
 ''
 'variable'
  'name'PgFilename'
  'default'postgresql-${PgVersion}.tar.gz'
 ''
 'variable'
  'name'PgBuildDir'
  'default'/usr/local/src'
 ''
 'variable'
  'name'PgUrl'
  'default'$PgBaseUrl$PgFilename'
 ''
 'variable'
  'name'PgUser'
  'default'postgres'
 ''
 'variable'
  'name'PgGroup'
  'default'web'
 ''
 'variable'
  'name'PgWithPgPort'
  'default'5432'
 ''
 'variable'
  'name'PgWithTcl'
  'default'no'
 ''
 'variable'
  'name'PgWithoutTk'
  'default'yes'
 ''
 'variable'
  'comment'Ask if tcl is preinstalled.'
  'name'TclPrefix'
  'default'/var/aolserver'
 ''
 'variable'
  'name'PgWithTclConfig'
  'default'$TclPrefix/lib'
 ''
 'variable'
  'name'PgWithMaxBackends'
  'default'32'
 ''
 'variable'
  'name'PgEnableDebug'
  'default'no'
 ''
 'command'groupadd $PgGroup'
 'command'useradd -g $PgGroup -d $PgPrefix $PgUser'
 'command'mkdir -p $PgPrefix
 'command'mkdir -p $PgBuildDir'
 'command'cd $PgBuildDir'
 'command'wget $PgUrl'
 'command'tar xvf $PgFilename'
 'command'cd postgresql-$PgVersion'
 'command'./configure --prefix=$PgPrefix \\
   --enable-locale \\
   --enable-mulitbyte=UNICODE \\
   --with-tcl=$PgWithTcl \\
   --without-tk=$PgWithoutTk \\
   --with-tclconfig=$PgWithTclConfig \\
   --enable-debug=$PgEnableDebug \\
   --enable-syslog=$PgEnableSyslog \\
   --with-pgport=$PgWithPgPort \\
   --with-maxbackends=$PgWithMaxBackends' 
 'command'su - $PgUser -c "echo \'export LD_LIBRARY_PATH=\$LD_LIBRARY_PATH:$PgPrefix/lib\' >> ~postgres/.bashrc"'
 'command'su - $PgUser -c "echo \'export PATH=\$PATH:$PgPrefix/bin\' >> ~postgres/.bashrc"'
 'command'make all'
 'command'make install'
 'command'su - postgres -c "$PgPrevix/bin/initdb -D $PgPrefix/data"'
 'command'su - postgres -c "createlang plpgsql template1"'
 'command'su - postgres -c "createlang -l template1"'
 'command'chown -R ${PgUser}.${PgGroup} $PgPrefix'
 'command'chmod 750 $PgPrefix'

 'comment'This is the end.'
''
