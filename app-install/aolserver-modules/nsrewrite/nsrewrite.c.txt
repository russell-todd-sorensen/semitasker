/*
 * nsrewrite.c --
 *
 *      A simple AOLserver module rewrite.
 *
 */

static const char *RCSID = "@(#) $Header: /cvsweb/nsd/nsrewrite/nsrewrite.c,v 1.1 2000/08/23 23:27:24 mayoff Exp $, compiled: " __DATE__ " " __TIME__;

#include "ns.h"


/*
 * The Ns_ModuleVersion variable is required.
 */
int Ns_ModuleVersion = 1;


/*
 * Private functions
 */
int
Ns_ModuleInit(char *hServer, char *hModule);

static int
RewriteInterpInit(Tcl_Interp *interp, void *context);

static int
RewriteURLCmd(ClientData context, Tcl_Interp *interp, int argc, char **argv);



/*
 *----------------------------------------------------------------------
 *
 * Ns_ModuleInit --
 *
 *      This is the nsrewrite module's entry point.  AOLserver runs
 *      this function right after the module is loaded.  It is used to
 *      read configuration data, initialize data structures, kick off
 *      the Tcl initialization function (if any), and do other things
 *      at startup.
 *
 * Results:
 *	NS_OK or NS_ERROR
 *
 * Side effects:
 *	Module loads and initializes itself.
 *
 *----------------------------------------------------------------------
 */
 
int
Ns_ModuleInit(char *hServer, char *hModule)
{

    return (Ns_TclInitInterps(hServer, RewriteInterpInit, NULL));

}



/*
 *----------------------------------------------------------------------
 *
 * RewriteInterpInit --
 *
 *      Register new commands with the Tcl interpreter.
 *
 * Results:
 *	NS_OK or NS_ERROR
 *
 * Side effects:
 *	A C function is registered with the Tcl interpreter.
 *
 *----------------------------------------------------------------------
 */
 
static int
RewriteInterpInit(Tcl_Interp *interp, void *context)
{

    Tcl_CreateCommand(interp, "ns_rewriteurl", RewriteURLCmd, NULL, NULL);

    return NS_OK;
}



/*
 *----------------------------------------------------------------------
 *
 * RewriteURLCmd --
 *
 *      A Tcl command that prints a friendly string with the name
 *      passed in on the first argument.
 *
 * Results:
 *	NS_OK
 *
 * Side effects:
 *	Tcl result is set to a string value.
 *
 *----------------------------------------------------------------------
 */
 
static int
RewriteURLCmd(ClientData context, Tcl_Interp *interp, int argc, char **argv)
{
    Ns_Conn *connPtr;
    
    if (argc != 2) {
	Tcl_AppendResult(interp, "wrong # of args: should be ", argv[0], " url", NULL);
	return TCL_ERROR;
    }

    connPtr = Ns_GetConn();
    if (connPtr == NULL) {
	Tcl_AppendResult(interp, "no current connection", NULL);
	return TCL_ERROR;
    }

    Ns_SetRequestUrl(connPtr->request, argv[1]);
    
    return NS_OK;
}
