
'application'
 'variable'
  'name'AolserverVersion'
  'default'4.0'
 ''
 'variable'
  'name'AolserverBaseUrl'
  'default'http://zmbh.com/app-install/aolserver/'
  'help'Directory URI for application file.'
 ''
 'variable'
  'name'AolserverFilename'
  'default'aolserver-${AolserverVersion}-src.tar.gz'
 ''
 'variable'
  'name'AolserverPrefix'
  'default'/usr/local/aolserver'
 ''
 'variable'
  'name'TclPrefix'
  'default'/usr/local/aolserver'
 ''
 'variable'
  'comment'Tcl must be installed prior to AOLserver.'
  'name'AolserverWithTcl'
  'default'$TclPrefix/lib'
  'help'Directory where tclConfig.sh is installed'
 ''
 'variable'
  'name'AolserverBuildDir'
  'default'/usr/local/src'
 ''
 'variable'
  'name'AolserverUrl'
  'default'$AolserverBaseUrl$AolserverFilename'
 ''
 'md5sum'3fb354dba28166a1004f9103553a3974'
 'command'cd $AolserverBuildDir'
 'command'wget $AolserverUrl'
 'command'tar xzf $AolserverFilename'
 'command'cd aolserver-${AolserverVersion}/'
 'command'./configure --prefix=$AolserverPrefix \\
  --with-tcl=$AolserverWithTcl'
 'command'make install'

 'comment'This is the end'
''
