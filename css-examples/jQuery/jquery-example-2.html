Customize UI Tabs:

  • Add or override styles, and
  • Changing the default .tabs(options) object properties.
<html>
 <head>
 
<style type="text/css" >
body { 
	background-color: #39C;
}
#my-tabs {
    background-color: transparent;
    margin: auto;
    margin-top: 20px;  
}

#my-tabs pre {
    background-color: silver;
    padding: 10px;
    border: 1px solid red;
}

#my-tabs pre.hilite {
    background-color: yellow;
    border: 1px dashed black;
}
.ui-widget-content {
    background-image: url(../images/linear-white-45diag-40px.png);
    background-size: cover;
}
</style>

<script language="javascript" type="text/javascript">

$(document).ready(function() {
	$('#my-tabs').tabs({
		add: {url:'#panel6',label:'jQuery',index:1 },
		ajaxOptions: null,
		cache: false,
		cookie: null, // e.g. { expires: 7, path: '/', domain: 'jquery.com', secure: true }
		collapsible: false,
		disable: null,
		disabled: [],
		enable: null,
		event: "click",
		fx: { height: 'toggle', opacity: 'toggle', duration: 200 },
		idPrefix: "ui-tabs-",
		load: null,
		panelTemplate: "<div></div>",
		remove: null,
		select: null,
		show: null,
		spinner: "<em>Loading…</em>",
		tabTemplate: "<li><a href='#{href}'><span>#{label}</span></a></li>"
	});
	
});

</script>
 

 </head>
  ....
